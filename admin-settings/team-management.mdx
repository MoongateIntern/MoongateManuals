> ## Managing Team Members 
<Steps>
    <Step title="Navigate to the 'Team Management' panel to see the list of Team Members details">
        Click 'Setting' on the sidebar, and then 'Team Management.'
        <img src="/images/admin-settings/team-management/team management panel.gif"/>
    </Step>
    <Step title="Add team members">
        <Warning>Only applicable to 'Owner' and 'Admin' roles team member</Warning>
        Click 'Add New', input the email address of the target team member, and then 'Add.'
        <img src="/images/admin-settings/team-management/add member.gif"/>
    </Step>
    <Step title="Deleting a team member">
        <Warning>Only applicable to 'Owner' roles team member</Warning>
        Click the trash bin icon next to the 'Role' column of the target team member under the 'Action' column.
        <img src="images/admin-settings/team-management/delete member.png"/>
    </Step>
</Steps>
