<Steps>
    <Step title="Click 'Check in'">
        Open the '[Pass](https://app.moongate.id)' page on the Moongate portal and click 'Check in'.
        <img src="/user-guide/images/the-basics/user-check-in-guide/check in button.png"/>
    </Step>
    <Step title="Present your QR code">
        Simply present the pop-up QR code to the event/membership check-in staff to scan. This QR code will work for 
        all of your events/memberships â€“ no need to select anything!
        <Tip>You can use Apple Wallet or Google Wallet to save the QR code to your phone.</Tip>
        <img src="/user-guide/images/the-basics/user-check-in-guide/website qr code.png"/>
        <Info>
            You will also have received a confirmation email when you purchased your ticket/membership pass. This email 
            contains a QR code for that ticket/membership pass that can also be used to check-in.
            <img src="/user-guide/images/the-basics/user-check-in-guide/email qr code.avif"/>
        </Info>
    </Step>
</Steps>
